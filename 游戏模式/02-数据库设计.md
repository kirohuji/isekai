# 数据库设计

## SQLite 完整建表语句

```sql
-- ============================================
-- 核心表
-- ============================================

-- 玩家状态
CREATE TABLE player (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL DEFAULT '未命名',
    gender TEXT DEFAULT '男',
    health INTEGER DEFAULT 100,
    max_health INTEGER DEFAULT 100,
    stamina INTEGER DEFAULT 100,
    max_stamina INTEGER DEFAULT 100,
    mental INTEGER DEFAULT 100,
    max_mental INTEGER DEFAULT 100,
    silver INTEGER DEFAULT 120,
    copper INTEGER DEFAULT 0,
    food_days INTEGER DEFAULT 7,
    medicine_count INTEGER DEFAULT 0,
    current_location_id INTEGER DEFAULT 1,
    current_time_block TEXT DEFAULT '清晨',
    current_date_year INTEGER DEFAULT 847,
    current_date_month INTEGER DEFAULT 9,
    current_date_day INTEGER DEFAULT 15,
    red_moon_countdown INTEGER DEFAULT 700,
    is_alive INTEGER DEFAULT 1,
    created_at TEXT DEFAULT (datetime('now')),
    FOREIGN KEY (current_location_id) REFERENCES locations(id)
);

-- 地点
CREATE TABLE locations (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    region TEXT,
    description TEXT,
    is_safe INTEGER DEFAULT 1,
    travel_cost INTEGER DEFAULT 1,
    connected_locations TEXT DEFAULT '[]'
);

-- 物品
CREATE TABLE inventory (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player_id INTEGER NOT NULL,
    item_name TEXT NOT NULL,
    item_type TEXT DEFAULT 'misc',
    quantity INTEGER DEFAULT 1,
    description TEXT,
    is_equipped INTEGER DEFAULT 0,
    FOREIGN KEY (player_id) REFERENCES player(id)
);

-- NPC
CREATE TABLE npcs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    race TEXT DEFAULT '人类',
    gender TEXT DEFAULT '男',
    location_id INTEGER,
    is_alive INTEGER DEFAULT 1,
    is_recruited INTEGER DEFAULT 0,
    description TEXT,
    FOREIGN KEY (location_id) REFERENCES locations(id)
);

-- 关系
CREATE TABLE relationships (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player_id INTEGER NOT NULL,
    npc_id INTEGER NOT NULL,
    affection INTEGER DEFAULT 0,
    trust INTEGER DEFAULT 0,
    status TEXT DEFAULT '陌生人',
    FOREIGN KEY (player_id) REFERENCES player(id),
    FOREIGN KEY (npc_id) REFERENCES npcs(id),
    UNIQUE(player_id, npc_id)
);

-- 势力声望
CREATE TABLE reputations (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player_id INTEGER NOT NULL,
    faction_name TEXT NOT NULL,
    reputation INTEGER DEFAULT 0,
    FOREIGN KEY (player_id) REFERENCES player(id),
    UNIQUE(player_id, faction_name)
);

-- 技能
CREATE TABLE skills (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player_id INTEGER NOT NULL,
    skill_name TEXT NOT NULL,
    level INTEGER DEFAULT 1,
    experience INTEGER DEFAULT 0,
    FOREIGN KEY (player_id) REFERENCES player(id),
    UNIQUE(player_id, skill_name)
);

-- ============================================
-- 事件系统
-- ============================================

-- 事件模板
CREATE TABLE event_templates (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    category TEXT NOT NULL,
    narrative_base TEXT NOT NULL,
    condition_json TEXT NOT NULL DEFAULT '{}',
    choices_json TEXT NOT NULL DEFAULT '[]',
    cooldown_days INTEGER DEFAULT 0,
    is_repeatable INTEGER DEFAULT 1,
    priority INTEGER DEFAULT 5
);

-- 事件日志
CREATE TABLE event_log (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player_id INTEGER NOT NULL,
    event_template_id INTEGER,
    event_name TEXT NOT NULL,
    choice_made TEXT,
    outcome TEXT,
    triggered_at_year INTEGER,
    triggered_at_month INTEGER,
    triggered_at_day INTEGER,
    triggered_at_time_block TEXT,
    FOREIGN KEY (player_id) REFERENCES player(id)
);

-- ============================================
-- 游戏状态
-- ============================================

-- 全局标记
CREATE TABLE flags (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player_id INTEGER NOT NULL,
    flag_name TEXT NOT NULL,
    flag_value TEXT,
    FOREIGN KEY (player_id) REFERENCES player(id),
    UNIQUE(player_id, flag_name)
);

-- 任务/目标
CREATE TABLE quests (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player_id INTEGER NOT NULL,
    quest_name TEXT NOT NULL,
    quest_description TEXT,
    quest_status TEXT DEFAULT '进行中',
    FOREIGN KEY (player_id) REFERENCES player(id)
);

-- 队伍成员
CREATE TABLE party (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player_id INTEGER NOT NULL,
    npc_id INTEGER NOT NULL,
    role TEXT DEFAULT '成员',
    FOREIGN KEY (player_id) REFERENCES player(id),
    FOREIGN KEY (npc_id) REFERENCES npcs(id)
);
```

## 数据库使用方式

游戏主持人（AI）在每轮交互中：

1. **读取**：`SELECT * FROM player WHERE id=1`——获取当前状态
2. **查询可选行动**：根据位置、时间、关系、标记生成选项
3. **执行行动**：更新 player 表和其他相关表
4. **检查事件触发**：`SELECT * FROM event_templates WHERE [条件满足]`
5. **触发事件**：将事件叙述 + 事件内的选择呈现给玩家
6. **记录日志**：`INSERT INTO event_log`
